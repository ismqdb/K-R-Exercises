#define libc_tsd_address(TYPE, KEY)   (&__libc_tsd_##KEY)

const unsigned short int **ctypebloc(void){
    return libc_tsd_address (const int32_t *, CTYPE_TOUPPER);
}

int isCtypeMy(char c, int type){
    return ((*ctypebloc())[(int) c] & (unsigned short int) type);
}

int IsBitMy(int bit){
	if(bit < 8)
		return ((1<<bit) << 8);
	else
		return ((1<<bit)>>8);
}

}
